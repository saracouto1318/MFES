\begin{vdmpp}[breaklines=true]
class Specialty

instance variables
  private name: Types`String;
  inv name <> [];
operations
(*@
\label{Specialty:7}
@*)
 public Specialty : Types`String ==> Specialty
  Specialty(n) == (name := n; return self)
 pre n <> []
 post name = n;
  
(*@
\label{getName:12}
@*)
 pure public getName : () ==> Types`String
  getName() == (return name);

end Specialty
\end{vdmpp}
