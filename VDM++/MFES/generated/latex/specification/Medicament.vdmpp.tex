\begin{vdmpp}[breaklines=true]
class Medicament

instance variables
  private name:Types`String;
  inv name <> [];
operations
(*@
\label{Medicament:7}
@*)
 public Medicament: Types`String ==> Medicament
  Medicament(n) == (name := n; return self)
 pre n <> []
 post name = n;
 
(*@
\label{getName:12}
@*)
 pure public getName: () ==> Types`String
  getName() == (return name);

end Medicament
